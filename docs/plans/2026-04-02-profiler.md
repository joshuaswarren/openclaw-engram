# Performance Profiler Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add an opt-in performance profiler that lets users understand timing of each Engram pipeline step (recall + extraction), see parallel vs blocking structure, and identify how their config/model choices impact responsiveness — all without external dependencies.

**Architecture:** A `ProfilingCollector` class captures `Date.now()` spans at key pipeline points, wraps `Promise.all` groups to record parallel wall-clock overlap, and writes JSONL trace files to `<memoryDir>/profiling/`. An MCP tool (`engram_profiling_report`) exposes aggregated stats. Zero new runtime dependencies — only `fs` and `Date.now()`.

**Tech Stack:** TypeScript, Node.js `fs`, existing `recall-qos.ts` metric recorder pattern

```typescript
import { mkdirSync, writeFileSync, readdirSync, unlinkSync, existsSync } from "node:fs";
import { join } from "node:path";
import { log } from "./logger.js";

// ── Types ────────────────────────────────────────────────────────────────────

export interface ProfileSpan {
  name: string;
  startOffsetMs: number;
  durationMs: number;
}

export interface ProfileParallelGroupMember {
  name: string;
  durationMs: number;
  resolvedIndex: number; // 0 = first to resolve
}

export interface ProfileParallelGroup {
  name: string;
  startOffsetMs: number;
  wallMs: number;
  members: ProfileParallelGroupMember[];
}

export interface ProfileTrace {
  ts: string;
  kind: "recall" | "extraction";
  traceId: string;
  sessionKey?: string;
  totalMs: number;
  spans: ProfileSpan[];
  parallelGroups?: ProfileParallelGroup[];
  configSnapshot?: Record<string, unknown>;
}

export interface ProfilingConfig {
  enabled: boolean;
  storageDir: string;
  maxTraces: number;
}

interface ActiveTrace {
  kind: "recall" | "extraction";
  traceId: string;
  sessionKey?: string;
  startMs: number;
  configSnapshot?: Record<string, unknown>;
  activeSpans: Map<string, number>; // name -> startMs (relative to trace start)
  completedSpans: ProfileSpan[];
  parallelGroups: ProfileParallelGroup[];
  traceStartMs: number; // absolute Date.now() when trace started
}

interface ParallelGroupHandle {
  name: string;
  startOffsetMs: number;
}

// ── Collector ───────────────────────────────────────────────────────────────

export class ProfilingCollector {
  private config: ProfilingConfig;
  private activeTrace: ActiveTrace | null = null;
  private recentTraces: ProfileTrace[] = [];
  private traceCounter = 0;

  constructor(config: ProfilingConfig) {
    this.config = config;
    if (config.enabled) {
      try {
        mkdirSync(config.storageDir, { recursive: true });
      } catch {
        log.debug(`profiling: could not create storage dir ${config.storageDir}`);
      }
    }
  }

  get isEnabled(): boolean {
    return this.config.enabled;
  }

  startTrace(kind: "recall" | "extraction", sessionKey?: string, configSnapshot?: Record<string, unknown>): void {
    if (!this.config.enabled) return;
    const traceId = `${kind}-${++this.traceCounter}-${Date.now().toString(36)}`;
    this.activeTrace = {
      kind,
      traceId,
      sessionKey,
      startMs: 0,
      configSnapshot,
      activeSpans: new Map(),
      completedSpans: [],
      parallelGroups: [],
      traceStartMs: Date.now(),
    };
  }

  startSpan(name: string): void {
    if (!this.activeTrace) return;
    this.activeTrace.activeSpans.set(name, Date.now() - this.activeTrace.traceStartMs);
  }

  endSpan(name: string): void {
    if (!this.activeTrace) return;
    const startOffset = this.activeTrace.activeSpans.get(name);
    if (startOffset === undefined) return;
    const now = Date.now() - this.activeTrace.traceStartMs;
    this.activeTrace.completedSpans.push({
      name,
      startOffsetMs: startOffset,
      durationMs: now - startOffset,
    });
    this.activeTrace.activeSpans.delete(name);
  }

  startParallelGroup(name: string): ParallelGroupHandle {
    if (!this.activeTrace) return { name, startOffsetMs: 0 };
    return {
      name,
      startOffsetMs: Date.now() - this.activeTrace.traceStartMs,
    };
  }

  async endParallelGroup(
    handle: ParallelGroupHandle,
    members: Array<{ name: string; promise: Promise<unknown> }>,
  ): Promise<void> {
    if (!this.activeTrace) {
      // Still await so we don't break the pipeline
      await Promise.all(members.map((m) => m.promise));
      return;
    }

    const raceStart = Date.now();
    const settled = await Promise.allSettled(
      members.map((m) => m.promise.then(() => Date.now() - raceStart)),
    );
    const wallMs = Date.now() - raceStart;
    const sortedWithIndex = settled
      .map((r, i) => ({
        index: i,
        durationMs: r.status === "fulfilled" ? (r.value as number) : -1,
        name: members[i].name,
      }))
      .sort((a, b) => a.durationMs - b.durationMs);

    this.activeTrace.parallelGroups.push({
      name: handle.name,
      startOffsetMs: handle.startOffsetMs,
      wallMs,
      members: sortedWithIndex.map((s, rank) => ({
        name: s.name,
        durationMs: s.durationMs,
        resolvedIndex: rank,
      })),
    });
  }

  endTrace(): ProfileTrace | null {
    if (!this.activeTrace) return null;
    const totalMs = Date.now() - this.activeTrace.traceStartMs;
    const trace: ProfileTrace = {
      ts: new Date().toISOString(),
      kind: this.activeTrace.kind,
      traceId: this.activeTrace.traceId,
      sessionKey: this.activeTrace.sessionKey,
      totalMs,
      spans: this.activeTrace.completedSpans,
      parallelGroups:
        this.activeTrace.parallelGroups.length > 0
          ? this.activeTrace.parallelGroups
          : undefined,
      configSnapshot: this.activeTrace.configSnapshot,
    };
    this.activeTrace = null;

    this.recentTraces.push(trace);
    if (this.recentTraces.length > this.config.maxTraces) {
      this.recentTraces.shift();
    }

    // Persist to JSONL
    this.persistTrace(trace);

    return trace;
  }

  getRecentTraces(limit: number = 10): ProfileTrace[] {
    return this.recentTraces.slice(-limit);
  }

  /** Compute aggregate stats from recent traces */
  getStats(): Record<string, { count: number; avgMs: number; p50Ms: number; p95Ms: number; maxMs: number }> {
    const stats: Record<string, { samples: number[] }> = {};
    for (const trace of this.recentTraces) {
      const key = trace.kind;
      if (!stats[key]) stats[key] = { samples: [] };
      stats[key].samples.push(trace.totalMs);

      for (const span of trace.spans) {
        const spanKey = `${trace.kind}.${span.name}`;
        if (!stats[spanKey]) stats[spanKey] = { samples: [] };
        stats[spanKey].samples.push(span.durationMs);
      }
    }
    const result: Record<string, { count: number; avgMs: number; p50Ms: number; p95Ms: number; maxMs: number }> = {};
    for (const [key, { samples }] of Object.entries(stats)) {
      if (samples.length === 0) continue;
      const sorted = [...samples].sort((a, b) => a - b);
      const p = (i: number) => sorted[Math.min(i, sorted.length - 1)];
      result[key] = {
        count: samples.length,
        avgMs: Math.round(samples.reduce((a, b) => a + b, 0) / samples.length),
        p50Ms: p(Math.floor(sorted.length * 0.5)),
        p95Ms: p(Math.floor(sorted.length * 0.95)),
        maxMs: sorted[sorted.length - 1],
      };
    }
    return result;
  }

  /** Identify the biggest bottleneck from recent traces */
  identifyBottleneck(): string | null {
    if (this.recentTraces.length === 0) return null;
    const latest = this.recentTraces[this.recentTraces.length - 1];
    if (latest.spans.length === 0) return null;
    const sorted = [...latest.spans].sort((a, b) => b.durationMs - a.durationMs);
    return sorted[0].name;
  }

  /** Compute parallel efficiency for a trace (how much wall time was wasted vs ideal) */
  parallelEfficiency(trace: ProfileTrace): number | null {
    if (!trace.parallelGroups || trace.parallelGroups.length === 0) return null;
    let totalIdealMs = 0;
    let totalWallMs = 0;
    for (const group of trace.parallelGroups) {
      totalIdealMs += group.members.reduce((sum, m) => sum + Math.max(0, m.durationMs), 0);
      totalWallMs += group.wallMs;
    }
    if (totalWallMs === 0) return 1;
    return Math.min(1, totalIdealMs / totalWallMs);
  }

  /** Clean up old trace files beyond maxTraces */
  pruneFiles(): void {
    if (!this.config.enabled || !existsSync(this.config.storageDir)) return;
    try {
      const files = readdirSync(this.config.storageDir)
        .filter((f) => f.endsWith(".jsonl"))
        .sort();
      while (files.length > this.config.maxTraces) {
        const oldest = files.shift();
        if (oldest) {
          unlinkSync(join(this.config.storageDir, oldest));
        }
      }
    } catch {
      // best-effort cleanup
    }
  }

  private persistTrace(trace: ProfileTrace): void {
    if (!this.config.enabled) return;
    try {
      const filename = `${trace.kind}-${trace.traceId}.jsonl`;
      writeFileSync(join(this.config.storageDir, filename), JSON.stringify(trace) + "\n");
      this.pruneFiles();
    } catch (err) {
      log.debug(`profiling: failed to persist trace: ${err}`);
    }
  }
}

// ── ASCII Formatter ──────────────────────────────────────────────────────────

export function formatProfileTraceAscii(trace: ProfileTrace): string {
  const lines: string[] = [];
  const totalSec = (trace.totalMs / 1000).toFixed(1);
  lines.push(`${trace.kind.charAt(0).toUpperCase() + trace.kind.slice(1)} Trace (${totalSec}s total)`);
  lines.push("─".repeat(60));

  // Find bottleneck
  const bottleneck = trace.spans.length > 0
    ? [...trace.spans].sort((a, b) => b.durationMs - a.durationMs)[0]
    : null;

  // Group spans by parallel groups for display
  const parallelGroupNames = new Set(trace.parallelGroups?.map((g) => g.name) ?? []);
  const maxDuration = Math.max(...trace.spans.map((s) => s.durationMs), 1);
  const barWidth = 40;

  for (const span of trace.spans) {
    const bar = "█".repeat(Math.max(1, Math.round((span.durationMs / maxDuration) * barWidth)));
    const durationStr = span.durationMs >= 1000
      ? `${(span.durationMs / 1000).toFixed(1)}s`
      : `${span.durationMs}ms`;
    const marker = bottleneck?.name === span.name ? " ← bottleneck" : "";
    lines.push(`  ${bar} ${span.name.padEnd(24)} ${durationStr.padStart(8)}${marker}`);
  }

  if (trace.parallelGroups && trace.parallelGroups.length > 0) {
    lines.push("");
    lines.push("Parallel Groups:");
    for (const group of trace.parallelGroups) {
      const wallStr = group.wallMs >= 1000
        ? `${(group.wallMs / 1000).toFixed(1)}s`
        : `${group.wallMs}ms`;
      const idealMs = group.members.reduce((s, m) => s + Math.max(0, m.durationMs), 0);
      const efficiency = group.wallMs > 0 ? Math.min(1, idealMs / group.wallMs) : 1;
      const effPct = Math.round(efficiency * 100);
      lines.push(`  [${group.name}] wall: ${wallStr}, efficiency: ${effPct}%`);
      for (const member of group.members) {
        const mStr = member.durationMs >= 1000
          ? `${(member.durationMs / 1000).toFixed(1)}s`
          : `${member.durationMs}ms`;
        lines.push(`    ├─ ${member.name.padEnd(24)} ${mStr.padStart(8)} (${member.resolvedIndex === 0 ? "fastest" : `#${member.resolvedIndex + 1}`})`);
      }
    }
  }

  return lines.join("\n");
}

        const traces = profiler.getRecentTraces(limit);
        const stats = profiler.getStats();
        const bottleneck = profiler.identifyBottleneck();

        if (format === "json") {
          return {
            content: [
              { type: "text", text: JSON.stringify({ traces, stats, bottleneck }, null, 2) },
            ],
          };
        }

        // ASCII format
        const lines: string[] = [];
        lines.push("Engram Profiling Report");
        lines.push("═".repeat(60));
        lines.push("");

        // Stats summary
        if (Object.keys(stats).length > 0) {
          lines.push("Aggregate Stats (recent traces):");
          for (const [key, s] of Object.entries(stats)) {
            lines.push(`  ${key}: avg=${s.avgMs}ms p50=${s.p50Ms}ms p95=${s.p95Ms}ms max=${s.maxMs}ms (n=${s.count})`);
          }
          lines.push("");
        }

        if (bottleneck) {
          lines.push(`Bottleneck: ${bottleneck}`);
          lines.push("");
        }

        // Per-trace output
        for (const trace of traces) {
          lines.push(formatProfileTraceAscii(trace));
          lines.push("");
        }

        return {
          content: [{ type: "text", text: lines.join("\n") }],
        };
      },
    );
  }
```
